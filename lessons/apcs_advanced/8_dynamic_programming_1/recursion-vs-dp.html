遞迴程式碼

function fibonacci(n) {
    if (n <= 2) return 1;
    return fibonacci(n-1) + fibonacci(n-2);
}

動態規劃程式碼

function fibonacciDP(n) {
    let dp = [0, 1, 1];
    for (let i = 3; i <= n; i++) {
        dp[i] = dp[i-1] + dp[i-2];
    }
    return dp[n];
}

// 每個 f(i) 只計算一次
