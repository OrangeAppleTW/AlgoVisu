// 組合枚舉演算法的Python實作
def enumerate_combinations(elements, m): n = len(elements) subset = [] def backtrack(start): if len(subset) == m: # 已選滿 m 個元素 print(subset) return # 剪枝：如果剩餘元素全部選取也無法湊滿 m if start >= n: return # 遍歷從 start 開始的每個候選元素 for i in range(start, n): subset.append(elements[i]) # 選取元素 i backtrack(i + 1) # 遞迴選取下一個之後的元素 subset.pop() # 回溯：移除剛選取的元素 backtrack(0) # 使用範例：C(3,2) enumerate_combinations([1, 2, 3], 2)
